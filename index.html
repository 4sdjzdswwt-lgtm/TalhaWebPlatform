// Lambanın açık veya kapalı olma durumunu tutan değişken let isLampOn = false; function pullLampChain() { const body = document.body; const startStep = document.getElementById('startStep'); const loginStep = document.getElementById('loginStep'); // Durumu tersine çevir (Açıksa kapat, kapalıysa aç) isLampOn = !isLampOn; if (isLampOn) { // Lambanın açık olduğunu belirten sınıfı body'ye ekle (CSS tetiklenecek) body.classList.add('light-on'); // Başlangıç yazısını gizle ve Giriş formunu göster if(startStep && loginStep) { startStep.style.display = 'none'; loginStep.style.display = 'block'; } } else { // Lambayı kapat body.classList.remove('light-on'); // Tekrar başlangıç ekranına dönmek isterseniz: if(startStep && loginStep) { startStep.style.display = 'block'; loginStep.style.display = 'none'; } } } // Formlar arası geçiş fonksiyonu (Giriş Yap / Kayıt Ol linkleri için) function switchStep(stepId) { // Tüm adımları gizle document.getElementById('loginStep').style.display = 'none'; document.getElementById('registerStep').style.display = 'none'; document.getElementById('verificationStep').style.display = 'none'; // İstenen adımı göster document.getElementById(stepId).style.display = 'block'; } // Şifre göster/gizle fonksiyonu function togglePasswordVisibility(inputId, button) { const input = document.getElementById(inputId); const icon = button.querySelector('i'); if (input.type === 'password') { input.type = 'text'; icon.className = 'fa-solid fa-eye'; } else { input.type = 'password'; icon.className = 'fa-solid fa-eye-slash'; } }
